Tarjan

Tarjan算法是一种离线算法离线算法}巧妙利用并查集维护祖先节点。
离线算法：将所有测试数据读入，然后进行算法，最后一起输出结果
在线算法：
将数据一个一个读入，在读入一个数据的时候进行算法，输出答案

算法步骤（“入下回离”）

  1. 从根开始深搜遍历，如u时，打标记
  2. 枚举u的儿子v，遍历完v的树回到u时，把v指向u
  3. 遍历完u的儿子，离u时，枚举以u为起点的查询，若终点v被搜过，则查找v的根，即u、v的LCA，答案记入ans[]
  4. 递归遍历完整棵树，得到全部查询答案

分析

时间复杂度 O(n+m)

代码

#include<iostream>
#include<algorithm>
#include<vector>
#define int long long
using namespace std;
const int maxn=5e5+10;
int n,m,s,a,b;
vector<int> e[maxn];
vector<pair<int,int>> query[maxn]; 
int fa[maxn],ans[maxn]; 
bool vis[maxn];
int find(int x){
   if(fa[x]==x) return fa[x];
   return fa[x]=find(fa[x]);
}
void tarjan(int u){
   vis[u]=1;
   for(auto i:e[u]){
   	if(!vis[i]){
   		tarjan(i);
   		fa[i]=u;
   	}
   }
   for(auto v:query[u]){
   	int q=v.first,p=v.second;
   	if(vis[q]){
   		ans[p]=find(q);
   	}
   }
}
signed main(){
   ios::sync_with_stdio(false),cin.tie(0),cout.tie(0);
   cin>>n>>m>>s;
   for(int i=1;i<n;i++){
   	cin>>a>>b;
   	e[a].push_back(b);
   	e[b].push_back(a);
   }
   for(int i=1;i<=m;i++){
   	cin>>a>>b;
   	query[a].push_back({b,i});
   	query[b].push_back({a,i});
   }
   for(int i=1;i<=maxn;i++){
   	fa[i]=i;
   }
   tarjan(s);
   for(int i=1;i<=m;i++){
   	cout<<ans[i]<<endl;
   }
   return 0;
}

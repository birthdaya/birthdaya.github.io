85分

得分

858585分，333个TLE

思路

使用大模拟，定义数组map[]map[]map[]存储颜色。如果opt==0opt==0opt==0，就把x行的值都设为ccc。如果opt==1opt==1opt==1，就把xxx列的值设为ccc。

代码

#include<iostream>
#define test ios::sync_with_stdio(false),cin.tie(0),cout.tie(0)
using namespace std;
const int maxn=1e5+10;
int t;
int n,m,q;
int opt,c,x;
signed main(){
//	freopen("1.txt","r",stdin); 
	test;
	cin>>t;
	while(t--){
		cin>>n>>m>>q;
		int map[n+5][m+5]={0};
		while(q--){
			cin>>opt>>x>>c;
			if(!opt){
				for(int i=1;i<=m;i++) map[x][i]=c;
			}else{
				for(int i=1;i<=n;i++) map[i][x]=c;
			}		
		}
		for(int i=1;i<=n;i++){
			for(int j=1;j<=m;j++){
				cout<<map[i][j]<<" ";
			}
			cout<<endl;
		}
	}
	return 0;
}

100分

思路

在858585分的基础上，添加上优先级。每一次操作修改的是一整行或者一整列，建立两个记录行列颜色的数组。为了解决颜色覆盖的问题我们可以记录该行颜色被填涂的优先级，记录顺序。如果被多次填涂同一行列覆盖即可。如果一个格子被所在行列同时覆盖，比较优先级。

代码

#include<iostream>
#include<string.h>
#define test ios::sync_with_stdio(false),cin.tie(0),cout.tie(0)
using namespace std;
const int maxn=1e5+10;
int f[5][maxn],map[5][maxn];//开二维是因为记录opt值，方便以后使用 
int t;
int n,m,q;
int opt,c,x;
signed main(){
//	freopen("1.txt","r",stdin); 
	test;
	cin>>t;
	while(t--){
		//初始化，要不然会去世的 
		memset(f,0,sizeof(f));
		memset(map,0,sizeof(map));
		cin>>n>>m>>q;
		for(int i=1;i<=q;i++){
			cin>>opt>>x>>c;
			map[opt][x]=c;//记录第x行或第x列的c值 
			f[opt][x]=i;//记录优先级，优先级定义为i，因为方便查询那个格子被覆盖（仅仅用于行列重叠时） 
		}
		for(int i=1;i<=n;i++){
			for(int j=1;j<=m;j++){
				if(f[0][i]>f[1][j]) cout<<map[0][i]<<" ";//如果这个格子在它的这一行和这一列都被刷过，就判断那个操作靠后 。如果 f[0][i]>f[1][j]，即0更新的比1更新的晚，就输出0更新的 
				else cout<<map[1][j]<<" ";//否则就直接输出1更新的 
			}
			cout<<endl;
		}
	}
	return 0;
}

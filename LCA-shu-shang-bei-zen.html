最近公共祖先定义

两个节点的最近公共祖先（LCA）就是这两个点的公共祖先里面，离他们最近的那个

倍增算法是最经典的求LCA算法
dep[u]dep[u]dep[u]存uuu点的深度
fa[u][i]fa[u][i]fa[u][i]存从uuu点向上跳222的iii次方层的祖先结点
例如：从节点 999 向上跳202^020层的祖先是3，向上跳212^121层的祖先是6，向上跳2^2层的祖先是1，向上跳 232^323 层的祖先是000（哨兵）

求解步骤

存储祖先

dfs一遍，创建ST稀疏表存储祖先节点
倍增递推公式：fa[u][i]=fa[fa[u][i−1]][i−1]fa[u][i]=fa[fa[u][i-1]][i-1]fa[u][i]=fa[fa[u][i−1]][i−1]

void dfs(int u,int f){
	dep[u]=dep[f]+1;
	fa[u][0]=f;
	for(int i=1;i<=25;i++){
		fa[u][i]=fa[fa[u][i-1]][i-1];
	}
	for(auto v:e[u]){
		if(v!=f) dfs(v,u);
	} 
}


将uuu和vvv跳到同一层

设uuu、vvv两点的深度差为yyy,将yyy进行二进制差分，可以将yyy此游标跳跃优化为“‘yyy的二进制表示所含111的个数”次游标跳跃一定能跳到同一层。

将uuu和vvv绑定一起跳到LCA的下一层

从最大的i开始循环尝试，一直尝试到0，最后游标u,v一定能停在LCA的下一层

   	if(fa[u][i]!=fa[v][i]){
         u=fa[u][i];
   	  v=fa[v][i];
   	}
     }
     return fa[u][0];

总结

  • dep[u]存存存u$点的深度
  • fa[u][i]fa[u][i]fa[u][i]存从uuu点向上跳222的iii次方层的祖先结点，
  • 打表：倍增递推，从小到大枚举
  • 查询：二进制拆分，从小到大枚举
  • 总的时间复杂度：O((m+n)logn)O((m+n)logn)O((m+n)logn)

代码

#include<iostream>
#include<algorithm>
#include<vector>
#define int long long
using namespace std;
const int maxn=5e5+10;
int n,m,s,a,b;
vector<int> e[maxn];
int dep[maxn],fa[maxn][50];
void dfs(int u,int f){
	dep[u]=dep[f]+1;
	fa[u][0]=f;
	for(int i=1;i<=25;i++){
		fa[u][i]=fa[fa[u][i-1]][i-1];
	}
	for(auto v:e[u]){
		if(v!=f) dfs(v,u);
	} 
}
int lca(int u,int v){
	if(dep[u]<dep[v]) swap(u,v);
	for(int i=25;i>=0;i--){
		if(dep[fa[u][i]]>=dep[v]) u=fa[u][i];
	}
	if(u==v) return v;
	for(int i=25;i>=0;i--){
		if(fa[u][i]!=fa[v][i]){
	      u=fa[u][i];
		  v=fa[v][i];
		}
	  }
	  return fa[u][0];
}

signed main(){
	ios::sync_with_stdio(false),cin.tie(0),cout.tie(0);
	cin>>n>>m>>s;
	for(int i=1;i<n;i++){
		cin>>a>>b;
		e[a].push_back(b);
		e[b].push_back(a);
	}
	dfs(s,0);
	for(int i=1;i<=m;i++){
		cin>>a>>b;
		cout<<lca(a,b)<<endl;
	}
	return 0;
}
